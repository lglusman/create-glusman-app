import { Checkbox } from '@mui/material'
import { use$entidadMayusculaPlural$Store } from '../../services/Stores/use$entidadMayusculaPlural$Store'

export const EsPaginado = () => {
  const setPagina = use$entidadMayusculaPlural$Store((state) => state.SetPagina)
  const pagina = use$entidadMayusculaPlural$Store((state) => state.Opciones.pag)

  function handleChange(_: React.ChangeEvent<HTMLInputElement>, checked: boolean): void {
    if (checked) {
      setPagina(1)
    } else {
      setPagina(0)
    }
  }

  return (
    <label>
      Paginado
      <Checkbox checked={(pagina || 0) > 0} onChange={handleChange} inputProps={{ 'aria-label': 'controlled' }} />
    </label>
  )
}
